
                {/* ── WATCHLIST ── */}
                <div className="p-4 border-b border-white/[0.06]">
                  <div className="flex items-center justify-between mb-3">
                    <span className="text-[9px] font-black uppercase tracking-[0.25em] text-zinc-500 flex items-center gap-1.5">
                      <Activity className="w-2.5 h-2.5 text-emerald-500" /> Watchlist
                    </span>
                    {selectedStock && (
                      <button
                        onClick={() => addToWatchlist(selectedStock)}
                        className="text-[8px] font-black text-indigo-400 hover:text-indigo-300 uppercase tracking-wider transition-colors bg-indigo-500/10 hover:bg-indigo-500/20 px-2 py-0.5 rounded-md border border-indigo-500/20"
                      >
                        + Watch
                      </button>
                    )}
                  </div>
                  {watchlist.length === 0 ? (
                    <div className="flex flex-col items-center justify-center py-4 gap-2">
                      <div className="w-8 h-8 rounded-xl bg-zinc-800/60 flex items-center justify-center">
                        <TrendingUp className="w-4 h-4 text-zinc-600" />
                      </div>
                      <p className="text-[8px] text-zinc-700 text-center leading-relaxed">Search a stock and click<br/>+ Watch to track it here</p>
                    </div>
                  ) : (
                    <div className="space-y-1">
                      {watchlist.map((w) => {
                        const isSelected = selectedStock?.key === w.key;
                        return (
                          <div
                            key={w.key}
                            className={cn(
                              "flex items-center justify-between px-3 py-2.5 rounded-xl cursor-pointer transition-all group relative overflow-hidden",
                              isSelected
                                ? "bg-indigo-500/12 border border-indigo-500/25"
                                : "hover:bg-white/[0.04] border border-transparent hover:border-white/[0.06]"
                            )}
                            onClick={() => { setSelectedStock(w); setQuery(w.symbol); fetchData(w); }}
                          >
                            {isSelected && <div className="absolute left-0 top-0 bottom-0 w-0.5 bg-indigo-500 rounded-r-full" />}
                            <div className="min-w-0 flex-1">
                              <div className="flex items-center gap-1.5">
                                <p className={cn("text-[11px] font-black tracking-tight", isSelected ? "text-indigo-300" : "text-zinc-200")}>{w.symbol}</p>
                                {isSelected && livePriceSource === 'upstox' && (
                                  <span className="w-1 h-1 rounded-full bg-emerald-400 animate-pulse" />
                                )}
                              </div>
                              <p className="text-[8px] text-zinc-600 truncate mt-0.5">{w.name.split(' ').slice(0, 3).join(' ')}</p>
                            </div>
                            <div className="flex items-center gap-2 ml-2">
                              {isSelected && livePrice != null && livePriceSource === 'upstox' ? (
                                <div className="text-right">
                                  <p className={cn("text-[10px] font-black font-mono transition-colors duration-150",
                                    livePriceFlash === 'up' ? "text-emerald-300" : livePriceFlash === 'down' ? "text-rose-300" : "text-white"
                                  )}>&#8377;{livePrice.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2})}</p>
                                  {livePriceChangePercent != null && (
                                    <p className={cn("text-[8px] font-bold font-mono", livePriceChangePercent >= 0 ? "text-emerald-400" : "text-rose-400")}>
                                      {livePriceChangePercent >= 0 ? '+' : ''}{livePriceChangePercent.toFixed(2)}%
                                    </p>
                                  )}
                                </div>
                              ) : isSelected && lastCandleClose != null ? (
                                <p className="text-[10px] font-mono text-zinc-500">&#8377;{lastCandleClose.toFixed(2)}</p>
                              ) : null}
                              <button
                                onClick={(e) => { e.stopPropagation(); setWatchlist(prev => prev.filter(x => x.key !== w.key)); }}
                                className="text-zinc-700 hover:text-rose-400 transition-colors opacity-0 group-hover:opacity-100 text-sm font-bold leading-none"
                              >x</button>
                            </div>
                          </div>
                        );
                      })}
                    </div>
                  )}
                </div>

                {/* ── SEARCH ── */}
                <div className="p-4 border-b border-white/[0.06]">
                  <span className="text-[9px] font-black uppercase tracking-[0.25em] text-zinc-500 block mb-2.5">Search Instrument</span>
                  <AssetSearch
                    query={query}
                    onQueryChange={setQuery}
                    onSelect={(s) => { setSelectedStock(s); addToWatchlist(s); setSuggestions([]); }}
                    containerRef={searchRef}
                  />
                </div>

                {/* ── TIMEFRAME ── */}
                <div className="p-4 border-b border-white/[0.06]">
                  <span className="text-[9px] font-black uppercase tracking-[0.25em] text-zinc-500 block mb-2.5">Timeframe</span>
                  <div className="flex gap-1 bg-black/40 p-1 rounded-xl border border-white/[0.06]">
                    {[
                      { val: '1minute', label: '1m' },
                      { val: '5minute', label: '5m' },
                      { val: '30minute', label: '30m' },
                      { val: 'day', label: '1D' },
                    ].map(({ val, label }) => (
                      <button
                        key={val}
                        onClick={() => setInterval(val)}
                        className={cn(
                          "flex-1 py-1.5 rounded-lg text-[9px] font-black uppercase tracking-wider transition-all",
                          interval === val
                            ? "bg-indigo-500 text-white shadow-lg shadow-indigo-500/30"
                            : "text-zinc-600 hover:text-zinc-300"
                        )}
                      >
                        {label}
                      </button>
                    ))}
                  </div>
                </div>

                {/* ── STRATEGY PRESETS ── */}
                <div className="p-4 border-b border-white/[0.06]">
                  <span className="text-[9px] font-black uppercase tracking-[0.25em] text-zinc-500 block mb-2.5">Strategy Presets</span>
                  <div className="grid grid-cols-2 gap-1.5">
                    {[
                      { label: 'Breakout', icon: 'B', preset: { recentBreakout: true, near52wHigh: true } },
                      { label: 'Intraday', icon: 'I', preset: { showRSI: true, showMACD: true } },
                      { label: 'Swing', icon: 'S', preset: { sma20AboveSma50: true, showVolAvg: true } },
                      { label: 'Smart $', icon: '$', preset: { sma50AboveSma200: true, showVolSpike: true } },
                    ].map(({ label, icon, preset }) => {
                      const isActive = Object.entries(preset).every(([k, v]) => advFilters[k as keyof FilterState] === v);
                      return (
                        <button
                          key={label}
                          onClick={() => setAdvFilters(isActive ? DEFAULT_FILTERS : { ...DEFAULT_FILTERS, ...preset })}
                          className={cn(
                            "px-2 py-2 rounded-xl border text-[9px] font-black uppercase tracking-wider transition-all flex items-center gap-1.5",
                            isActive
                              ? "bg-indigo-500/15 border-indigo-500/35 text-indigo-300 shadow-[0_0_12px_rgba(99,102,241,0.1)]"
                              : "bg-black/20 border-white/[0.06] text-zinc-600 hover:border-white/10 hover:text-zinc-300"
                          )}
                        >
                          <span className="w-4 h-4 rounded-md bg-white/5 flex items-center justify-center text-[8px]">{icon}</span> {label}
                        </button>
                      );
                    })}
                  </div>
                </div>

                {/* ── OVERLAYS ── */}
                <div className="p-4 border-b border-white/[0.06]">
                  <span className="text-[9px] font-black uppercase tracking-[0.25em] text-zinc-500 block mb-2.5">Chart Overlays</span>
                  <div className="flex gap-2">
                    <button
                      onClick={() => setShowSMA20(!showSMA20)}
                      className={cn(
                        "flex-1 py-2 rounded-xl border text-[9px] font-black uppercase tracking-wider transition-all",
                        showSMA20 ? "bg-emerald-500/10 border-emerald-500/30 text-emerald-400" : "bg-black/20 border-white/[0.06] text-zinc-600 hover:text-zinc-300"
                      )}
                    >
                      SMA 20
                    </button>
                    <button
                      onClick={() => setShowSMA50(!showSMA50)}
                      className={cn(
                        "flex-1 py-2 rounded-xl border text-[9px] font-black uppercase tracking-wider transition-all",
                        showSMA50 ? "bg-orange-500/10 border-orange-500/30 text-orange-400" : "bg-black/20 border-white/[0.06] text-zinc-600 hover:text-zinc-300"
                      )}
                    >
                      SMA 50
                    </button>
                  </div>
                </div>

                {/* ── DATE RANGE ── */}
                <div className="p-4 border-b border-white/[0.06]">
                  <span className="text-[9px] font-black uppercase tracking-[0.25em] text-zinc-500 block mb-2.5">Date Range</span>
                  <div className="space-y-2">
                    <input
                      type="date"
                      className="w-full bg-black/40 border border-white/[0.08] rounded-xl py-2 px-3 text-[11px] text-zinc-300 focus:ring-1 focus:ring-indigo-500/40 outline-none transition-all"
                      value={fromDate}
                      onChange={(e) => setFromDate(e.target.value)}
                    />
                    <input
                      type="date"
                      className="w-full bg-black/40 border border-white/[0.08] rounded-xl py-2 px-3 text-[11px] text-zinc-300 focus:ring-1 focus:ring-indigo-500/40 outline-none transition-all"
                      value={toDate}
                      onChange={(e) => setToDate(e.target.value)}
                    />
                  </div>
                </div>

                {/* ── ADVANCED FILTERS ── */}
                <div className="p-4 border-b border-white/[0.06]">
                  <AnalyticsFilters filters={advFilters} onChange={setAdvFilters} />
                </div>

                {/* ── EXECUTE ── */}
                <div className="p-4">
                  <button
                    onClick={() => fetchData()}
                    disabled={loading || !selectedStock}
                    className="w-full bg-gradient-to-r from-indigo-600 to-violet-600 hover:from-indigo-500 hover:to-violet-500 disabled:from-zinc-800 disabled:to-zinc-800 disabled:text-zinc-600 text-white font-black py-3.5 rounded-xl shadow-lg shadow-indigo-500/20 transition-all flex items-center justify-center gap-2 group active:scale-[0.98]"
                  >
                    {loading ? <Loader2 className="w-4 h-4 animate-spin" /> : <TrendingUp className="w-4 h-4 group-hover:scale-110 transition-transform" />}
                    <span className="text-[10px] uppercase tracking-[0.2em]">Load Chart</span>
                  </button>
                </div>

              </div>
            </section>
          </div>
